        const loginForm = document.getElementById('loginForm');
        const errorMsg = document.getElementById('errorMsg');

        // Demo user validation
        function validateLogin(email, password) {
            const validEmail = 'admin@skywings.com';
            const validPassword = 'admin123';
            return email === validEmail && password === validPassword;
        }

        loginForm.addEventListener('submit', (e) => {
            e.preventDefault();
            
            const email = document.getElementById('email').value.trim();
            const password = document.getElementById('password').value;

            if (validateLogin(email, password)) {
                // Store login session
                sessionStorage.setItem('isLoggedIn', 'true');
                sessionStorage.setItem('userEmail', email);
                sessionStorage.setItem('loginTime', new Date().toISOString());
                
                // Redirect to dashboard
                window.location.href = 'dashboard.html';
            } else {
                errorMsg.style.display = 'block';
                setTimeout(() => {
                    errorMsg.style.display = 'none';
                }, 3000);
            }
        });

        // Check if already logged in
        if (sessionStorage.getItem('isLoggedIn') === 'true') {
            window.location.href = 'dashboard.html';
        }
    </script>

        // Check login status
        function checkAuth() {
            const isLoggedIn = sessionStorage.getItem('isLoggedIn');
            if (isLoggedIn !== 'true') {
                window.location.href = 'index.html';
            }
        }

        // Load user info
        function loadUserInfo() {
            const email = sessionStorage.getItem('userEmail') || 'admin@skywings.com';
            const name = email.split('@')[0];
            document.getElementById('userName').innerHTML = name.charAt(0).toUpperCase() + name.slice(1);
            document.getElementById('userEmail').innerHTML = email;
            document.getElementById('welcomeName').innerHTML = name.charAt(0).toUpperCase() + name.slice(1);
        }

        // Logout function
        function logout() {
            sessionStorage.clear();
            window.location.href = 'index.html';
        }

        // ============================================
        // IFRAME FUNCTIONS

        function refreshIframe() {
            const iframe = document.getElementById('cognosIframe');

            iframe.src = src;
        }

        function openFullscreen() {
            const iframe = document.getElementById('cognosIframe');
            if (iframe.requestFullscreen) {
                iframe.requestFullscreen();
            } else if (iframe.webkitRequestFullscreen) {
                iframe.webkitRequestFullscreen();
            }
        }
